#### 快速排序

```cpp
//cpp

void quick_sort(int* A, int lo, int hi) {
    if (lo + 1 >= hi) return;
    int x = A[(lo + hi) >> 1];
    int L = lo, R = hi - 1;
    
    while (1) {
        while (A[L] < x) ++L;
        while (A[R] > x) --R;
        if (L < R) {
            int tmp = A[L];
            A[L] = A[R];
            A[R] = A[L];
            ++L;
            --R;
        } else
            break;
    }
    quick_sort(A, lo, L);
    quick_sort(A, L, hi);
}
```
